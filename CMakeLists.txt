cmake_minimum_required(VERSION 3.0)

set(CMAKE_CXX_STANDARD 11)

include(cmake/HunterGate.cmake)

HunterGate(
    URL "https://github.com/poljkee2010/hunter/archive/v0.19.137.1.tar.gz"
    SHA1 "c85118ae943500d5fce2afb24cee4253eb3296ae"
)

project(main)
hunter_add_package(CURL)
find_package(CURL CONFIG REQUIRED)

add_executable(main ${CMAKE_CURRENT_SOURCE_DIR}/main.cpp) # Добавляем цель main (исполняемый файл) main по исходному файлу main.cpp 
target_link_libraries(main CURL::libcurl OpenSSL::SSL) # Линковка цели main с целью libcurl из пространства имён CURL и цели SSL,Crypto из пространства имён OpenSSL
